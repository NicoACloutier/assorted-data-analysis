langs <- read.csv("data/cleaned/langs.csv")
for (fam in unique(langs$family)) {
    p <- langs %>% 
        filter(family == fam) %>%
        ggplot(aes(x=year, y=log(TotalBothSexes), color=language)) +
            geom_line() +
            ggtitle(fam)
    print(p)
}
